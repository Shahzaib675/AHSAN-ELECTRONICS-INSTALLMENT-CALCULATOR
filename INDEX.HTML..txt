<script>
function roundToNearest5(num) {
    return Math.round(num / 5) * 5;
}

function formatPKR(num) {
    return 'PKR ' + num.toLocaleString();
}

function calculateInstallment() {
    const price = parseFloat(document.getElementById('price').value) || 0;
    const advance = parseFloat(document.getElementById('advance').value) || 0;
    const marginPercent = parseFloat(document.getElementById('margin').value) || 0;

    if (price <= 0) {
        document.getElementById('result').innerHTML = "<b>Please enter a valid price.</b>";
        return;
    }

    const marginAmount = (price * marginPercent) / 100;
    const totalPrice = price + marginAmount;
    const remaining = totalPrice - advance;
    let monthly = remaining / 12;

    monthly = roundToNearest5(monthly);

    document.getElementById('result').innerHTML = `
        <b>Price:</b> ${formatPKR(price)}<br>
        <b>Margin Amount:</b> ${formatPKR(marginAmount)}<br>
        <b>Total Price (with margin):</b> ${formatPKR(totalPrice)}<br>
        <b>Advance Paid:</b> ${formatPKR(advance)}<br>
        <b>Remaining Amount:</b> ${formatPKR(remaining)}<br>
        <b>Monthly Installment (12 months):</b> <span style="color:green">${formatPKR(monthly)}</span>
    `;
}
</script>
